矢量Vector

矢量(Vector)

定义:

x <- c(1,2,3,4)
assign("x",c(1,2,3,4))
c(1,2,3,4) -> x

矢量运算(Vector arithmetic )

x <- c(1,2,3,4)
y <- c(5,6,7,8)
v <- 2*x+y+1 // v: 8 11 14 17
